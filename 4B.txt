DECLARE
sno student.sid%type;
name student.sname%type; srank student.rank%type;

sno := &sno;
name := '&name';
srank := '&srank';

dbms_output.put_line('One record inserted');
COMMIT;
-- adding savepoint SAVEPOINT s1;
-- second time asking user for input sno := &sno;
name := '&name';
srank := '&srank';
INSERT into student values(sno,name,srank);
dbms_output.put_line('One record inserted');
ROLLBACK TO SAVEPOINT s1;
END;
